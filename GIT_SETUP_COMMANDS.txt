# Git Setup va Push Commands
# PowerShell-da bajaring:

# 1. Git Initialize (bajarilgan)
git init

# 2. Git Config
git config user.name "OptoMarketDev"
git config user.email "dev@optommarket.uz"

# 3. Add all files
git add .

# 4. Commit
git commit -m "Initial commit: Full-stack OptoMarket.uz"

# 5. Add remote
git remote add origin https://github.com/optimbazar-ai/Optom_market.git

# 6. Branch rename
git branch -M main

# 7. Push
git push -u origin main

# 8. Verify
git remote -v

# git push -u origin main --force
